9/// Guesses the content type based on example data. If the function is
10/// uncertain, @result_uncertain will be set to [`true`]. Either @filename
11/// or @data may be [`None`], in which case the guess will be based solely
12/// on the other argument.
13/// ## `filename`
14/// a path, or [`None`]
15/// ## `data`
16/// a stream of data, or [`None`]
17///
18/// # Returns
19///
20/// a string indicating a guessed content type for the
21///     given data. Free with g_free()
22///
23/// ## `result_uncertain`
24/// return location for the certainty
25///     of the result, or [`None`]
26#[doc(alias = "g_content_type_guess")]
27pub fn content_type_guess<'a>(
28    filename: Option<impl AsRef<std::path::Path>>,
29    data: impl Into<Option<&'a [u8]>>,
30) -> (glib::GString, bool) {
31    let data = data.into();
32    let data_size = data.map_or(0, |d| d.len());
33    unsafe {
34        let mut result_uncertain = std::mem::MaybeUninit::uninit();
35        let ret = from_glib_full(ffi::g_content_type_guess(
36            filename.as_ref().map(|p| p.as_ref()).to_glib_none().0,
37            data.map_or(ptr::null(), |d| d.to_glib_none().0),
38            data_size,
39            result_uncertain.as_mut_ptr(),
40        ));
41        (ret, from_glib(result_uncertain.assume_init()))
42    }
43}
