209#[doc(alias = "g_mkdtemp")]
210pub fn mkdtemp(tmpl: impl AsRef<std::path::Path>) -> Option<std::path::PathBuf> {
211    unsafe {
212        // NOTE: This modifies the string in place and returns it but does not free it
213        // if it returns NULL.
214        let tmpl = tmpl.as_ref().to_glib_full();
215        let res = ffi::g_mkdtemp(tmpl);
216        if res.is_null() {
217            ffi::g_free(tmpl as ffi::gpointer);
218            None
219        } else {
220            from_glib_full(res)
221        }
222    }
223}

302/// ## `filename`

304///
305/// # Returns
306///
307/// [`true`] on success, [`false`] if an error occurred
308///

310/// location to store an allocated string, use g_free() to free
311///     the returned string
312// rustdoc-stripper-ignore-next-stop
313/// Reads an entire file into allocated memory, with good error
314/// checking.
315///
316/// If the call was successful, it returns [`true`] and sets @contents to the file
317/// contents and @length to the length of the file contents in bytes. The string
318/// stored in @contents will be nul-terminated, so for text files you can pass
319/// [`None`] for the @length argument. If the call was not successful, it returns
320/// [`false`] and sets @error. The error domain is `G_FILE_ERROR`. Possible error
321/// codes are those in the #GFileError enumeration. In the error case,
322/// @contents is set to [`None`] and @length is set to zero.
323/// ## `filename`
324/// name of a file to read contents from, in the GLib file name encoding
325///
326/// # Returns
327///
328/// [`true`] on success, [`false`] if an error occurred
329///
330/// ## `contents`
331/// location to store an allocated string, use g_free() to free
332///     the returned string
333#[doc(alias = "g_file_get_contents")]
334pub fn file_get_contents(
335    filename: impl AsRef<std::path::Path>,
336) -> Result<crate::Slice<u8>, crate::Error> {
337    unsafe {
338        let mut contents = ptr::null_mut();
339        let mut length = mem::MaybeUninit::uninit();
340        let mut error = ptr::null_mut();
341        let _ = ffi::g_file_get_contents(
342            filename.as_ref().to_glib_none().0,
343            &mut contents,
344            length.as_mut_ptr(),
345            &mut error,
346        );
347        if error.is_null() {
348            Ok(crate::Slice::from_glib_full_num(
349                contents,
350                length.assume_init() as _,
351            ))
352        } else {
353            Err(from_glib_full(error))
354        }
355    }
356}

358pub fn is_canonical_pspec_name(name: &str) -> bool {
359    name.as_bytes().iter().enumerate().all(|(i, c)| {
360        i != 0 && (*c >= b'0' && *c <= b'9' || *c == b'-')
361            || (*c >= b'A' && *c <= b'Z')
362            || (*c >= b'a' && *c <= b'z')
363    })
364}

428    //Mutex to prevent run environment tests parallel
429    fn lock() -> &'static Mutex<()> {
430        static LOCK: OnceLock<Mutex<()>> = OnceLock::new();
431        LOCK.get_or_init(|| Mutex::new(()))
432    }
433
434    const VAR_NAME: &str = "function_environment_test";
435
436    fn check_getenv(val: &str) {
437        let _data = lock().lock().unwrap();
438
439        env::set_var(VAR_NAME, val);
440        assert_eq!(env::var_os(VAR_NAME), Some(val.into()));
441        assert_eq!(crate::getenv(VAR_NAME), Some(val.into()));
442
443        let environ = crate::environ();
444        assert_eq!(crate::environ_getenv(&environ, VAR_NAME), Some(val.into()));
445    }
446
447    fn check_setenv(val: &str) {
448        let _data = lock().lock().unwrap();
449
450        crate::setenv(VAR_NAME, val, true).unwrap();
451        assert_eq!(env::var_os(VAR_NAME), Some(val.into()));
452    }

471        let uri: GString = "file:///foo/bar.txt".into();
472        if let Ok((filename, hostname)) = crate::filename_from_uri(&uri) {
473            assert_eq!(filename, PathBuf::from(r"/foo/bar.txt"));
474            assert_eq!(hostname, None);

476            unreachable!();

480        if let Ok((filename, hostname)) = crate::filename_from_uri(&uri) {

482            assert_eq!(hostname, Some(GString::from("host")));
483        } else {
484            unreachable!();
485        }
