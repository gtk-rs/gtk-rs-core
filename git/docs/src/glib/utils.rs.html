70#[doc(alias = "g_mkdtemp")]
71pub fn mkdtemp(tmpl: impl AsRef<std::path::Path>) -> Option<std::path::PathBuf> {
72    unsafe {
73        // NOTE: This modifies the string in place and returns it but does not free it
74        // if it returns NULL.
75        let tmpl = tmpl.as_ref().to_glib_full();
76        let res = ffi::g_mkdtemp(tmpl);
77        if res.is_null() {
78            ffi::g_free(tmpl as ffi::gpointer);
79            None
80        } else {
81            from_glib_full(res)
82        }
83    }
84}

123/// Reads an entire file into allocated memory, with good error
124/// checking.
125///
126/// If the call was successful, it returns [`true`] and sets @contents to the file
127/// contents and @length to the length of the file contents in bytes. The string
128/// stored in @contents will be nul-terminated, so for text files you can pass
129/// [`None`] for the @length argument. If the call was not successful, it returns
130/// [`false`] and sets @error. The error domain is `G_FILE_ERROR`. Possible error
131/// codes are those in the #GFileError enumeration. In the error case,
132/// @contents is set to [`None`] and @length is set to zero.
133/// ## `filename`
134/// name of a file to read contents from, in the GLib file name encoding
135///
136/// # Returns
137///
138/// [`true`] on success, [`false`] if an error occurred
139///
140/// ## `contents`
141/// location to store an allocated string, use g_free() to free
142///     the returned string
143#[doc(alias = "g_file_get_contents")]
144pub fn file_get_contents(
145    filename: impl AsRef<std::path::Path>,
146) -> Result<crate::Slice<u8>, crate::Error> {
147    unsafe {
148        let mut contents = ptr::null_mut();
149        let mut length = mem::MaybeUninit::uninit();
150        let mut error = ptr::null_mut();
151        let _ = ffi::g_file_get_contents(
152            filename.as_ref().to_glib_none().0,
153            &mut contents,
154            length.as_mut_ptr(),
155            &mut error,
156        );
157        if error.is_null() {
158            Ok(crate::Slice::from_glib_full_num(
159                contents,
160                length.assume_init() as _,
161            ))
162        } else {
163            Err(from_glib_full(error))
164        }
165    }
166}

168pub fn is_canonical_pspec_name(name: &str) -> bool {
169    name.as_bytes().iter().enumerate().all(|(i, c)| {
170        i != 0 && (*c >= b'0' && *c <= b'9' || *c == b'-')
171            || (*c >= b'A' && *c <= b'Z')
172            || (*c >= b'a' && *c <= b'z')
173    })
174}

178    unescaped: impl IntoGStr,
179    reserved_chars_allowed: Option<impl IntoGStr>,
180    allow_utf8: bool,
181) -> crate::GString {
182    unescaped.run_with_gstr(|unescaped| {
183        reserved_chars_allowed.run_with_gstr(|reserved_chars_allowed| unsafe {
184            from_glib_full(ffi::g_uri_escape_string(
185                unescaped.to_glib_none().0,
186                reserved_chars_allowed.to_glib_none().0,
187                allow_utf8.into_glib(),
188            ))
189        })
190    })
191}

236    use std::{env, sync::Mutex, sync::OnceLock};

238    //Mutex to prevent run environment tests parallel
239    fn lock() -> &'static Mutex<()> {
240        static LOCK: OnceLock<Mutex<()>> = OnceLock::new();
241        LOCK.get_or_init(|| Mutex::new(()))
242    }
243
244    const VAR_NAME: &str = "function_environment_test";
245
246    fn check_getenv(val: &str) {
247        let _data = lock().lock().unwrap();
248
249        env::set_var(VAR_NAME, val);
250        assert_eq!(env::var_os(VAR_NAME), Some(val.into()));
251        assert_eq!(crate::getenv(VAR_NAME), Some(val.into()));
252
253        let environ = crate::environ();
254        assert_eq!(crate::environ_getenv(&environ, VAR_NAME), Some(val.into()));
255    }
256
257    fn check_setenv(val: &str) {
258        let _data = lock().lock().unwrap();
259
260        crate::setenv(VAR_NAME, val, true).unwrap();
261        assert_eq!(env::var_os(VAR_NAME), Some(val.into()));
262    }

276    #[test]
277    fn test_filename_from_uri() {
278        use std::path::PathBuf;
279
280        use crate::GString;
281        let uri: GString = "file:///foo/bar.txt".into();
282        if let Ok((filename, hostname)) = crate::filename_from_uri(&uri) {
283            assert_eq!(filename, PathBuf::from(r"/foo/bar.txt"));
284            assert_eq!(hostname, None);
285        } else {
286            unreachable!();
287        }
288
289        let uri: GString = "file://host/foo/bar.txt".into();
290        if let Ok((filename, hostname)) = crate::filename_from_uri(&uri) {
291            assert_eq!(filename, PathBuf::from(r"/foo/bar.txt"));
292            assert_eq!(hostname, Some(GString::from("host")));
293        } else {
294            unreachable!();
295        }
296    }

298    #[test]
299    fn test_uri_parsing() {
300        use crate::GString;
301        assert_eq!(
302            crate::uri_parse_scheme("foo://bar"),
303            Some(GString::from("foo"))
304        );
305        assert_eq!(crate::uri_parse_scheme("foo"), None);
306
307        let escaped = crate::uri_escape_string("&foo", crate::NONE_STR, true);
308        assert_eq!(escaped, GString::from("%26foo"));
309
310        let unescaped = crate::uri_unescape_string(escaped.as_str(), crate::GStr::NONE);
311        assert_eq!(unescaped, Some(GString::from("&foo")));
312
313        assert_eq!(
314            crate::uri_unescape_segment(Some("/foo"), crate::NONE_STR, crate::NONE_STR),
315            Some(GString::from("/foo"))
316        );
317        assert_eq!(
318            crate::uri_unescape_segment(Some("/foo%"), crate::NONE_STR, crate::NONE_STR),
319            None
320        );
321    }
