6    /// An interface that handles the lifecycle of dynamically loaded types.
7    ///
8    /// The GObject type system supports dynamic loading of types.
9    /// It goes as follows:
10    ///
11    /// 1. The type is initially introduced (usually upon loading the module
12    ///  the first time, or by your main application that knows what modules
13    ///  introduces what types), like this:
14    ///  **⚠️ The following code is in c ⚠️**
15    ///
16    /// ```c
17    ///    new_type_id = g_type_register_dynamic (parent_type_id,
18    ///                                           "TypeName",
19    ///                                           new_type_plugin,
20    ///                                           type_flags);
21    ///    ```
22    ///  where `new_type_plugin` is an implementation of the
23    ///  `GTypePlugin` interface.
24    ///
25    /// 2. The type's implementation is referenced, e.g. through
26    ///  [func[`Object`][crate::Object].ref] or through [func[`Object`][crate::Object]]
27    ///  (this is being called by [ctor[`Object`][crate::Object].new]) or through one of the above
28    ///  done on a type derived from `new_type_id`.
29    ///
30    /// 3. This causes the type system to load the type's implementation by calling
31    ///  [method[`Object`][crate::Object].use] and [method[`Object`][crate::Object].complete_type_info]
32    ///  on `new_type_plugin`.
33    ///
34    /// 4. At some point the type's implementation isn't required anymore, e.g. after
35    ///  [method[`Object`][crate::Object].unref] or [func[`Object`][crate::Object]]
36    ///  (called when the reference count of an instance drops to zero).
37    ///
38    /// 5. This causes the type system to throw away the information retrieved
39    ///  from [method[`Object`][crate::Object].complete_type_info] and then it calls
40    ///  [method[`Object`][crate::Object].unuse] on `new_type_plugin`.
41    ///
42    /// 6. Things may repeat from the second step.
43    ///
44    /// So basically, you need to implement a `GTypePlugin` type that
45    /// carries a use_count, once use_count goes from zero to one, you need
46    /// to load the implementation to successfully handle the upcoming
47    /// [method[`Object`][crate::Object].complete_type_info] call. Later, maybe after
48    /// succeeding use/unuse calls, once use_count drops to zero, you can
49    /// unload the implementation again. The type system makes sure to call
50    /// [method[`Object`][crate::Object].use] and [method[`Object`][crate::Object].complete_type_info]
51    /// again when the type is needed again.
52    ///
53    /// [class[`Object`][crate::Object]] is an implementation of `GTypePlugin` that
54    /// already implements most of this except for the actual module loading and
55    /// unloading. It even handles multiple registered types per module.
56    ///
57    /// # Implements
58    ///
59    /// [`TypePluginExt`][trait@crate::prelude::TypePluginExt]

127/// Trait containing all [`struct@TypePlugin`] methods.
128///
129/// # Implementors
130///
131/// [`TypeModule`][struct@crate::TypeModule], [`TypePlugin`][struct@crate::TypePlugin]
132// rustdoc-stripper-ignore-next-stop
133/// Trait containing all [`struct@TypePlugin`] methods.
134///
135/// # Implementors
136///
137/// [`TypeModule`][struct@crate::TypeModule], [`TypePlugin`][struct@crate::TypePlugin]
138pub trait TypePluginExt: IsA<TypePlugin> + 'static {
139    /// Calls the `complete_interface_info` function from the
140    /// `GTypePluginClass` of `self`. There should be no need to use this
141    /// function outside of the GObject type system itself.
142    /// ## `instance_type`
143    /// the `GType` of an instantiatable type to which the interface
144    ///  is added
145    /// ## `interface_type`
146    /// the `GType` of the interface whose info is completed
147    /// ## `info`
148    /// the [`InterfaceInfo`][crate::InterfaceInfo] to fill in
149    // rustdoc-stripper-ignore-next-stop
150    /// Calls the `complete_interface_info` function from the
151    /// `GTypePluginClass` of `self`. There should be no need to use this
152    /// function outside of the GObject type system itself.
153    /// ## `instance_type`
154    /// the `GType` of an instantiatable type to which the interface
155    ///  is added
156    /// ## `interface_type`
157    /// the `GType` of the interface whose info is completed
158    /// ## `info`
159    /// the [`InterfaceInfo`][crate::InterfaceInfo] to fill in
160    #[doc(alias = "g_type_plugin_complete_interface_info")]
161    fn complete_interface_info(
162        &self,
163        instance_type: crate::types::Type,
164        interface_type: crate::types::Type,
165    ) -> InterfaceInfo {
166        let info = InterfaceInfo::default();
167        unsafe {
168            gobject_ffi::g_type_plugin_complete_interface_info(
169                self.as_ref().to_glib_none().0,
170                instance_type.into_glib(),
171                interface_type.into_glib(),
172                info.as_ptr(),
173            );
174        }
175        info
176    }
177
178    /// Calls the `complete_type_info` function from the `GTypePluginClass` of `self`.
179    /// There should be no need to use this function outside of the GObject
180    /// type system itself.
181    /// ## `g_type`
182    /// the `GType` whose info is completed
183    /// ## `info`
184    /// the [`TypeInfo`][crate::TypeInfo] struct to fill in
185    /// ## `value_table`
186    /// the [`TypeValueTable`][crate::TypeValueTable] to fill in
187    // rustdoc-stripper-ignore-next-stop
188    /// Calls the `complete_type_info` function from the `GTypePluginClass` of `self`.
189    /// There should be no need to use this function outside of the GObject
190    /// type system itself.
191    /// ## `g_type`
192    /// the `GType` whose info is completed
193    /// ## `info`
194    /// the [`TypeInfo`][crate::TypeInfo] struct to fill in
195    /// ## `value_table`
196    /// the [`TypeValueTable`][crate::TypeValueTable] to fill in
197    #[doc(alias = "g_type_plugin_complete_type_info")]
198    fn complete_type_info(&self, g_type: crate::types::Type) -> (TypeInfo, TypeValueTable) {
199        let info = TypeInfo::default();
200        let value_table = TypeValueTable::default();
201        unsafe {
202            gobject_ffi::g_type_plugin_complete_type_info(
203                self.as_ref().to_glib_none().0,
204                g_type.into_glib(),
205                info.as_ptr(),
206                value_table.as_ptr(),
207            );
208        }
209        (info, value_table)
210    }
211
212    /// Calls the `unuse_plugin` function from the `GTypePluginClass` of
213    /// `self`. There should be no need to use this function outside of
214    /// the GObject type system itself.
215    // rustdoc-stripper-ignore-next-stop
216    /// Calls the `unuse_plugin` function from the `GTypePluginClass` of
217    /// `self`. There should be no need to use this function outside of
218    /// the GObject type system itself.
219    #[doc(alias = "g_type_plugin_unuse")]
220    fn unuse(&self) {
221        unsafe {
222            gobject_ffi::g_type_plugin_unuse(self.as_ref().to_glib_none().0);
223        }
224    }
225
226    /// Calls the `use_plugin` function from the `GTypePluginClass` of
227    /// `self`. There should be no need to use this function outside of
228    /// the GObject type system itself.
229    // rustdoc-stripper-ignore-next-stop
230    /// Calls the `use_plugin` function from the `GTypePluginClass` of
231    /// `self`. There should be no need to use this function outside of
232    /// the GObject type system itself.
233    #[doc(alias = "g_type_plugin_use")]
234    #[doc(alias = "use")]
235    fn use_(&self) {
236        unsafe {
237            gobject_ffi::g_type_plugin_use(self.as_ref().to_glib_none().0);
238        }
239    }
240}
