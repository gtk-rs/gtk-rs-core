use glib::prelude::*;
use glib::types::{Pointee, Pointer};
use std::ptr::NonNull;

let pointer = NonNull::<Pointee>::dangling();
let value = pointer.to_value();
assert!(value.is::<Pointer>());
assert_eq!(value.get(), Ok(pointer.as_ptr()));
